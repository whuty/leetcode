class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        if(nums.empty())
            return -1;
        int n = nums.size();
        int l = 0;
        int r = n - 1;
        int m;
        while(l <= r) {
            m = (l + r) * 0.5;
            if(nums[m] < target) l = m + 1;
            else if(nums[m] > target) r = m - 1;
            else return m;
        }
        return r + 1;
    }
};
